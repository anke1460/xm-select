## 修改提示



### 修改选项提示文字

:::demo 
```html
<div id="demo1" class="xm-select-demo"></div>

<script>
var demo1 = xmSelect.render({
	el: '#demo1', 
	tips: '你喜欢什么水果呢?',
	data: [
		{name: '水果', value: 1},
		{name: '蔬菜', value: 2},
		{name: '桌子', value: 3},
		{name: '北京', value: 4},
	]
})
</script>
```
:::


### 修改空数据提示文字

:::demo 
```html
<div id="demo2" class="xm-select-demo"></div>

<script>
var demo2 = xmSelect.render({
	el: '#demo2', 
	empty: '呀, 没有数据呢',
	data: [ ]
})
</script>
```
:::
